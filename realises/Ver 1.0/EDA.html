max_price = data_plot['Цена'].max()
mean_price = data_plot['Цена'].mean()

key_prices = [min_price, mean_price, max_price]

plt.xticks(x_ticks, [f'{x:,.0f}' for x in x_ticks], rotation=45)

plt.show()
